367 
368  //DIE_TEMP_RDY interrupt must be enabled
369  //See issue 19: https://github.com/sparkfun/SparkFun_MAX3010x_Sensor_Library/issues/19
370 
371  // Step 1: Config die temperature register to take 1 temperature sample
372  writeRegister8(_i2caddr, MAX30105_DIETEMPCONFIG, 0x01);

374  // Poll for bit to clear, reading is then complete
375  // Timeout after 100ms
376 
377  unsigned long startTime = millis();
378  while (millis() - startTime < 100)
379  {
380  uint8_t response1 = readRegister8(_i2caddr, MAX30105_DIETEMPCONFIG); //Original way
381  if ((response1 & 0x01) == 0) break; //We're done!
382 
383  //Check to see if DIE_TEMP_RDY interrupt is set
384  uint8_t response = readRegister8(_i2caddr, MAX30105_INTSTAT2);
385  if ((response & MAX30105_INT_DIE_TEMP_RDY_ENABLE) > 0) break; //We're done!
386  delay(1); //Let's not over burden the I2C bus
387  }
388  //TODO How do we want to fail? With what type of error?
389  //? if(millis() - startTime >= 100) return(-999.0);
390 
391  long t1=micros();
392  // Step 2: Read die temperature register (integer)
393  int8_t tempInt = readRegister8(_i2caddr, MAX30105_DIETEMPINT);
394  uint8_t tempFrac = readRegister8(_i2caddr, MAX30105_DIETEMPFRAC); //Causes the clearing of the DIE_TEMP_RDY interrupt
395 
396 
397  // Step 3: Calculate temperature (datasheet pg. 23)
398  return (float)tempInt + ((float)tempFrac * 0.0625);

432 //Setup the sensor
433 //The MAX30105 has many settings. By default we select:
434 // Sample Average = 4
435 // Mode = MultiLED
436 // ADC Range = 16384 (62.5pA per LSB)

438 //Use the default setup if you are just getting started with the MAX30105 sensor
439 void MAX30105::setup(byte powerLevel, byte sampleAverage, byte ledMode, int sampleRate, int pulseWidth, int adcRange) {
440  softReset(); //Reset all configuration, threshold, and data registers to POR values
441 
442  //FIFO Configuration
443  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
444  //The chip will average multiple samples of same type together if you wish
445  if (sampleAverage == 1) setFIFOAverage(MAX30105_SAMPLEAVG_1); //No averaging per FIFO record
446  else if (sampleAverage == 2) setFIFOAverage(MAX30105_SAMPLEAVG_2);
447  else if (sampleAverage == 4) setFIFOAverage(MAX30105_SAMPLEAVG_4);
448  else if (sampleAverage == 8) setFIFOAverage(MAX30105_SAMPLEAVG_8);
449  else if (sampleAverage == 16) setFIFOAverage(MAX30105_SAMPLEAVG_16);
450  else if (sampleAverage == 32) setFIFOAverage(MAX30105_SAMPLEAVG_32);
451  else setFIFOAverage(MAX30105_SAMPLEAVG_4);
452 
453  //setFIFOAlmostFull(2); //Set to 30 samples to trigger an 'Almost Full' interrupt
454  enableFIFORollover(); //Allow FIFO to wrap/roll over
455  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
456 
457  //Mode Configuration
458  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
459  if (ledMode == 3) setLEDMode(MAX30105_MODE_MULTILED); //Watch all three LED channels
460  else if (ledMode == 2) setLEDMode(MAX30105_MODE_REDIRONLY); //Red and IR
461  else setLEDMode(MAX30105_MODE_REDONLY); //Red only
462  activeLEDs = ledMode; //Used to control how many bytes to read from FIFO buffer
463  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
464 
465  //Particle Sensing Configuration
466  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
467  if(adcRange < 4096) setADCRange(MAX30105_ADCRANGE_2048); //7.81pA per LSB
468  else if(adcRange < 8192) setADCRange(MAX30105_ADCRANGE_4096); //15.63pA per LSB
469  else if(adcRange < 16384) setADCRange(MAX30105_ADCRANGE_8192); //31.25pA per LSB
470  else if(adcRange == 16384) setADCRange(MAX30105_ADCRANGE_16384); //62.5pA per LSB
471  else setADCRange(MAX30105_ADCRANGE_2048);
472 
473  if (sampleRate < 100) setSampleRate(MAX30105_SAMPLERATE_50); //Take 50 samples per second
474  else if (sampleRate < 200) setSampleRate(MAX30105_SAMPLERATE_100);
475  else if (sampleRate < 400) setSampleRate(MAX30105_SAMPLERATE_200);
476  else if (sampleRate < 800) setSampleRate(MAX30105_SAMPLERATE_400);
477  else if (sampleRate < 1000) setSampleRate(MAX30105_SAMPLERATE_800);
478  else if (sampleRate < 1600) setSampleRate(MAX30105_SAMPLERATE_1000);
479  else if (sampleRate < 3200) setSampleRate(MAX30105_SAMPLERATE_1600);
480  else if (sampleRate == 3200) setSampleRate(MAX30105_SAMPLERATE_3200);
481  else setSampleRate(MAX30105_SAMPLERATE_50);
482 
483  //The longer the pulse width the longer range of detection you'll have
484  //At 69us and 0.4mA it's about 2 inches
485  //At 411us and 0.4mA it's about 6 inches
486  if (pulseWidth < 118) setPulseWidth(MAX30105_PULSEWIDTH_69); //Page 26, Gets us 15 bit resolution
487  else if (pulseWidth < 215) setPulseWidth(MAX30105_PULSEWIDTH_118); //16 bit resolution
488  else if (pulseWidth < 411) setPulseWidth(MAX30105_PULSEWIDTH_215); //17 bit resolution
489  else if (pulseWidth == 411) setPulseWidth(MAX30105_PULSEWIDTH_411); //18 bit resolution
490  else setPulseWidth(MAX30105_PULSEWIDTH_69);
491  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
492 
493  //LED Pulse Amplitude Configuration
494  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
495  //Default is 0x1F which gets us 6.4mA
496  //powerLevel = 0x02, 0.4mA - Presence detection of ~4 inch
497  //powerLevel = 0x1F, 6.4mA - Presence detection of ~8 inch
498  //powerLevel = 0x7F, 25.4mA - Presence detection of ~8 inch
499  //powerLevel = 0xFF, 50.0mA - Presence detection of ~12 inch

593 //Polls the sensor for new data
594 //Call regularly
595 //If new data is available, it updates the head and tail in the main struct
596 //Returns number of new samples obtained
597 uint16_t MAX30105::check(void)
598 {
599  //Read register FIDO_DATA in (3-byte * number of active LED) chunks
600  //Until FIFO_RD_PTR = FIFO_WR_PTR
601 
602  byte readPointer = getReadPointer();
603  byte writePointer = getWritePointer();
604 
605  int numberOfSamples = 0;
606 
607  //Do we have new data?
608  if (readPointer != writePointer)
609  {
610  //Calculate the number of readings we need to get from sensor
611  numberOfSamples = writePointer - readPointer;
612  if (numberOfSamples < 0) numberOfSamples += 32; //Wrap condition
613 
614  //We now have the number of readings, now calc bytes to read
615  //For this example we are just doing Red and IR (3 bytes each)
616  int bytesLeftToRead = numberOfSamples * activeLEDs * 3;
617 
618  //Get ready to read a burst of data from the FIFO register
619  _i2cPort->beginTransmission(MAX30105_ADDRESS);
620  _i2cPort->write(MAX30105_FIFODATA);
621  _i2cPort->endTransmission();
622 
623  //We may need to read as many as 288 bytes so we read in blocks no larger than I2C_BUFFER_LENGTH
624  //I2C_BUFFER_LENGTH changes based on the platform. 64 bytes for SAMD21, 32 bytes for Uno.
625  //Wire.requestFrom() is limited to BUFFER_LENGTH which is 32 on the Uno
626  while (bytesLeftToRead > 0)
627  {
628  int toGet = bytesLeftToRead;
629  if (toGet > I2C_BUFFER_LENGTH)
630  {
631  //If toGet is 32 this is bad because we read 6 bytes (Red+IR * 3 = 6) at a time
632  //32 % 6 = 2 left over. We don't want to request 32 bytes, we want to request 30.
633  //32 % 9 (Red+IR+GREEN) = 5 left over. We want to request 27.
634 
635  toGet = I2C_BUFFER_LENGTH - (I2C_BUFFER_LENGTH % (activeLEDs * 3)); //Trim toGet to be a multiple of the samples we need to read
636  }
637 
638  bytesLeftToRead -= toGet;
639 
640  //Request toGet number of bytes from sensor
641  _i2cPort->requestFrom(MAX30105_ADDRESS, toGet);
642 
643  while (toGet > 0)
644  {
645  sense.head++; //Advance the head of the storage struct
646  sense.head %= STORAGE_SIZE; //Wrap condition
647 
648  byte temp[sizeof(uint32_t)]; //Array of 4 bytes that we will convert into long
649  uint32_t tempLong;
650 
651  //Burst read three bytes - RED
652  temp[3] = 0;
653  temp[2] = _i2cPort->read();
654  temp[1] = _i2cPort->read();
655  temp[0] = _i2cPort->read();
656 
657  //Convert array to long
658  memcpy(&tempLong, temp, sizeof(tempLong));
659 
660  tempLong &= 0x3FFFF; //Zero out all but 18 bits
661 
662  sense.red[sense.head] = tempLong; //Store this reading into the sense array
663 
664  if (activeLEDs > 1)
665  {
666  //Burst read three more bytes - IR
667  temp[3] = 0;
668  temp[2] = _i2cPort->read();
669  temp[1] = _i2cPort->read();
670  temp[0] = _i2cPort->read();
671 
672  //Convert array to long
673  memcpy(&tempLong, temp, sizeof(tempLong));
674 
675  tempLong &= 0x3FFFF; //Zero out all but 18 bits
676 
677  sense.IR[sense.head] = tempLong;
678  }
679 
680  if (activeLEDs > 2)
681  {
682  //Burst read three more bytes - Green
683  temp[3] = 0;
684  temp[2] = _i2cPort->read();
685  temp[1] = _i2cPort->read();
686  temp[0] = _i2cPort->read();
687 
688  //Convert array to long
689  memcpy(&tempLong, temp, sizeof(tempLong));
690 
691  tempLong &= 0x3FFFF; //Zero out all but 18 bits
692 
693  sense.green[sense.head] = tempLong;
694  }
695 
696  toGet -= activeLEDs * 3;
697  }
698 
699  } //End while (bytesLeftToRead > 0)
700 
701  } //End readPtr != writePtr
702 
703  return (numberOfSamples); //Let the world know how much new data we found
704 }
