407 uint32_t getSerialNumber_ID(void) {
408 
409  volatile uint32_t val1, val2, val3, val4;
410  volatile uint32_t *ptr1 = (volatile uint32_t *)0x0080A00C; // SAMD21only
411  val1 = *ptr1;
412  volatile uint32_t *ptr = (volatile uint32_t *)0x0080A040; // SAMD21only

414  ptr++;
415  val3 = *ptr;
416  ptr++;
417  val4 = *ptr;
418 
419  uint32_t SerialNumber = val1+val2+val3+val4;
420  return SerialNumber;
